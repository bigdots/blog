# 深拷贝


```js
/**
 * deepCopy
 * @param {*} data 要拷贝的数据
 */
function deepCopy(data) {
    var buf;
    if (data instanceof Array) {
        buf = []; // 创建一个空的数组
        var i = data.length;
        while (i--) {
            buf[i] = deepCopy(data[i]);
        }
        return buf;
    } else if (data instanceof Object) {
        buf = {}; // 创建一个空对象
        for (var k in data) {
            // 为这个对象添加新的属性
            buf[k] = deepCopy(data[k]);
        }
        return buf;
    } else {
        return data;
    }
}
```

tips：`for in` 遍历数组会遍历 Array.prototype 上的可枚举属性。可以使用 `hasOwnProperty` 解决。
