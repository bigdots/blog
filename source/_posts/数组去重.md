# 数组去重

ES6 简单写法

```js
Array.prototype.uniq = function() {
    return [...new Set(this)];
};
```

ES5 写法

```js
Array.prototype.uniq = function() {
    var hash = {};
    return this.reduce((result, current) => {
        if (!hash[current]) {
            hash[current] = true;
            result.push(current);
        }
        return result;
    }, []);
}
```

```js
Array.prototype.uniq = function() {
    // 长度只有1，直接返回当前的拷贝
    if (this.length <= 1) {
        return this.slice(0);
    }
    var buf = [];
    for (var i = 0, l = this.length; i < l; i++) {
        if (!_isExist(buf, this[i])) {
            buf.push(this[i]);
        }
    }
    return buf; // 判断是否重复
    function _isExist(bufArray, target) {
        if (bufArray.length === 0) {
            return false;
        }
        var bufItem;
        for (var i = 0, l = bufArray.length; i < l; i++) {
            bufItem = bufArray[i];
            if (bufItem === target) {
                return true;
                break;
            }
            // NaN 需要特殊处理
            if (
                typeof target === "number" &&
                typeof bufItem === "number" &&
                isNaN(bufItem) &&
                isNaN(target)
            ) {
                return true;
            }
        }
        return false;
    }
};
```
